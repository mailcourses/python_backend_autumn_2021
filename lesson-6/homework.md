как пример фукция на получение списка объектов

    @require_GET
    def get_post_list(request):
    	posts = Post.objects.filter(is_published=True)
    	data = [
    		{
    			'id': post.id,
    			'title': post.title,
    			'text': post.text,
    			'date': post.created_at
    		} for post in posts
    	]
    	return JsonResponse({'posts': data})
